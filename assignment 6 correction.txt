function Kiosk () {

var fruits = [];

function setFruit (fruit) {
fruits.push(fruit);
}

function getFruits () {
return fruits;
}

var publicAPI = {
setFruit : setFruit,
getFruits : getFruits
}

return publicAPI;

}

var kiosk = Kiosk();
kiosk.setFruit("kiwi");
kiosk.setFruit("mangoes");
kiosk.setFruit("bananas");
console.log(kiosk.getFruits());
